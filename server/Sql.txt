CREATE TABLE MM_USER_RATING (RATING_ID BIGINT NOT NULL GENERATED BY DEFAULT AS IDENTITY, USER_ID VARCHAR(255), USER_RATING BIGINT, USER_PROJECT BIGINT, USER_TEAM BIGINT, USER_COMMENT VARCHAR(255), USER_RATED_TIMESTAMP DATE DEFAULT CURRENT_DATE,USER_RATED_TIME TIMESTAMP DEFAULT CURRENT_TIMESTAMP,  PRIMARY KEY(RATING_ID));
